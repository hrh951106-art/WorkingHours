```typescript
private enrichFieldWithType(field: any, customFields: any[]) {
  // CUSTOM字段：从CustomField中查找数据源
  if (field.fieldType === 'CUSTOM') {
    const customField = customFields.find((cf) => cf.code === field.fieldCode);
    return {
      ...field,
      type: customField?.type || 'TEXT',
      dataSource: customField?.dataSource || null,
    };
  }

  // SYSTEM字段：也尝试从CustomField中查找数据源（新增逻辑）
  if (field.fieldType === 'SYSTEM') {
    const systemFieldsNeedingDataSource = [
      'gender', 'nation', 'maritalStatus', 'politicalStatus',
      'marital_status', 'political_status', 'educationLevel', 'education_level',
      'employeeType', 'employee_type'
    ];

    if (systemFieldsNeedingDataSource.includes(field.fieldCode)) {
      const customField = customFields.find((cf) =>
        cf.code === field.fieldCode || cf.code === field.fieldCode.toLowerCase()
      );

      if (customField?.dataSource) {
        return {
          ...field,
          type: 'SELECT',
          dataSource: customField.dataSource,
        };
      }
    }
  }

  return field;
}
```

- ✅ 向后兼容，不影响现有功能
- ✅ 支持驼峰和下划线两种命名

## 🥈 **方案2: 前端直接查询数据源API**

### 实现思路
前端在获取页签配置后，对需要数据源的字段，直接调用数据源API获取选项

### 前端代码示例
```typescript
// 前端服务代码
async getTabFieldWithDataSource(fieldCode: string) {
  // 1. 获取字段配置
  const field = await api.get(`/api/hr/employee-info-tabs/fields/${fieldCode}`);

  // 2. 如果字段需要数据源，直接查询
  if (['gender', 'nation', 'maritalStatus'].includes(field.fieldCode)) {
    const dataSourceCode = this.mapFieldToDataSource(field.fieldCode);
    const dataSource = await api.get(`/api/hr/data-sources/${dataSourceCode}`);
    return {
      ...field,
      dataSource: dataSource
    };
  }

  return field;
}

private mapFieldToDataSource(fieldCode: string): string {
  const mapping = {
    'gender': 'GENDER',
    'nation': 'NATION',
    'maritalStatus': 'MARITAL_STATUS',
    'politicalStatus': 'POLITICAL_STATUS',
    'educationLevel': 'EDUCATION',
    'employeeType': 'EMPLOYEE_TYPE'
  };
  return mapping[fieldCode] || fieldCode.toUpperCase();
}
```

### 优点
- ✅ 不需要修改后端代码
- ✅ 前端灵活控制
- ✅ 可以实现缓存优化

### 缺点
- ❌ 需要维护字段到数据源的映射关系
- ❌ 多次API请求

## 🥉 **方案3: 在EmployeeInfoTabField添加dataSourceId字段**

### 数据库修改
```sql
-- 1. 添加dataSourceId字段到EmployeeInfoTabField表
ALTER TABLE "EmployeeInfoTabField"
ADD COLUMN "dataSourceId" INTEGER;

-- 2. 添加外键约束
ALTER TABLE "EmployeeInfoTabField"
ADD CONSTRAINT "EmployeeInfoTabField_dataSourceId_fkey"
FOREIGN KEY ("dataSourceId") REFERENCES "DataSource"(id) ON DELETE SET NULL;

-- 3. 创建索引
CREATE INDEX idx_field_datasource
ON "EmployeeInfoTabField"("dataSourceId");

-- 4. 为需要数据源的字段设置dataSourceId
UPDATE "EmployeeInfoTabField" f
SET "dataSourceId" = (
    SELECT d.id FROM "DataSource" d
    WHERE d.code = CASE f.fieldCode
        WHEN 'gender' THEN 'GENDER'
        WHEN 'nation' THEN 'NATION'
        WHEN 'maritalStatus' THEN 'MARITAL_STATUS'
        WHEN 'politicalStatus' THEN 'POLITICAL_STATUS'
        WHEN 'educationLevel' THEN 'EDUCATION'
        WHEN 'employeeType' THEN 'EMPLOYEE_TYPE'
        ELSE UPPER(f.fieldCode)
    END
)
WHERE f.fieldCode IN ('gender', 'nation', 'maritalStatus', 'politicalStatus', 'educationLevel', 'employeeType');
```

### 后端代码修改

## 🏆 **方案4: 创建SQL视图（最快，无需改代码）**

### 创建视图
```sql
CREATE OR REPLACE VIEW "EmployeeInfoTabFieldWithDataSource" AS
SELECT
    f.*,
    cf.id AS "customFieldId",
    cf.type AS "customFieldType",
    cf."dataSourceId",
    d.code AS "dataSourceCode",
    d.name AS "dataSourceName",
    d.type AS "dataSourceType"
FROM "EmployeeInfoTabField" f
LEFT JOIN "CustomField" cf ON cf.code = f.fieldCode
LEFT JOIN "DataSource" d ON d.id = cf."dataSourceId";
```

### 查询使用
```typescript
// 后端代码中使用视图查询
async getTabsForDisplay() {
  const tabs = await this.prisma.$queryRaw`
    SELECT
      t.*,
      g.*,
      f.*,
      fds."dataSourceCode",
      fds.options
    FROM "EmployeeInfoTab" t
    LEFT JOIN "EmployeeInfoTabGroup" g ON g."tabId" = t.id
    LEFT JOIN "EmployeeInfoTabField" f ON f."tabId" = t.id AND (f."groupId" IS NULL OR f."groupId" = g.id)
    LEFT JOIN "EmployeeInfoTabFieldWithDataSource" fds ON fds.id = f.id
    LEFT JOIN LATERAL (
      SELECT json_agg(
        json_build_object(
          'id', o.id,
          'label', o.label,
          'value', o.value,
          'sort', o.sort
        ) ORDER BY o.sort
      ) AS options
      FROM "DataSourceOption" o
      WHERE o."dataSourceId" = fds."dataSourceId"
        AND o."isActive" = true
    ) fds_options ON true
    WHERE t.status = 'ACTIVE'
    ORDER BY t.sort, g.sort, f.sort
  `;

  return tabs;
}
```

### 优点
- ✅ 无需修改代码，只需创建视图
- ✅ 即时生效，无需重启应用
- ✅ 可以包含所有关联数据

### 缺点
- ❌ 原生SQL查询，失去Prisma类型安全
- ❌ 需要手动维护视图定义

## ⚡ **最快速的解决方案（立竿见影）**

### 临时修复SQL（5分钟生效）

```sql
-- ====== 快速修复：直接在查询中关联CustomField ======

-- 创建一个查询函数或视图，包含数据源信息
CREATE OR REPLACE FUNCTION get_employee_info_tabs_with_datasource()
RETURNS TABLE (
    "tabId" INTEGER,
    "tabCode" VARCHAR,
    "tabName" VARCHAR,
    "groupId" INTEGER,
    "groupCode" VARCHAR,
    "groupName" VARCHAR,
    "fieldId" INTEGER,
    "fieldCode" VARCHAR,
    "fieldName" VARCHAR,
    "fieldType" VARCHAR,
    "dataSourceCode" VARCHAR,
    "dataSourceName" VARCHAR,
    "options" JSONB
) AS $$
BEGIN
    RETURN QUERY
    WITH tab_fields AS (
        SELECT
            t.id AS "tabId",
            t.code AS "tabCode",
            t.name AS "tabName",
            g.id AS "groupId",
            g.code AS "groupCode",
            g.name AS "groupName",
            f.id AS "fieldId",
            f.fieldCode,
            f.fieldName,
            f.fieldType,
            cf."dataSourceId",
            d.code AS "dataSourceCode",
            d.name AS "dataSourceName"
        FROM "EmployeeInfoTab" t
        LEFT JOIN "EmployeeInfoTabGroup" g ON g."tabId" = t.id
        LEFT JOIN "EmployeeInfoTabField" f ON f."tabId" = t.id AND (f."groupId" IS NULL OR f."groupId" = g.id)
        LEFT JOIN "CustomField" cf ON cf.code = f.fieldCode
        LEFT JOIN "DataSource" d ON d.id = cf."dataSourceId"
        WHERE t.status = 'ACTIVE'
    )
    SELECT
        tf.*,
        (
            SELECT json_agg(json_build_object(
                'id', o.id,
                'label', o.label,
                'value', o.value,
                'sort', o.sort
            ) ORDER BY o.sort)
            FROM "DataSourceOption" o
            WHERE o."dataSourceId" = tf."dataSourceId"
              AND o."isActive" = true
        ) AS options
    FROM tab_fields tf
    ORDER BY tf."tabCode", tf."groupCode", tf.sort;
END;
$$ LANGUAGE plpgsql;

-- 使用函数查询
SELECT * FROM get_employee_info_tabs_with_datasource();
```
