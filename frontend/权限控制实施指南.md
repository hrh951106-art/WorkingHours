# 前端权限控制实施指南

## 权限控制Hook

`/frontend/src/hooks/usePermissions.ts` 提供了以下hooks：

### 1. usePermissions()
获取用户的所有功能权限列表

```typescript
const permissions = usePermissions();
// 返回: ['hr:org:view', 'hr:emp:view', ...]
```

### 2. useHasPermission(permission: string)
检查用户是否拥有指定权限

```typescript
const canView = useHasPermission('hr:employee:view');
// 返回: true 或 false
```

### 3. useHasAnyPermission(permissions: string[])
检查用户是否拥有指定权限中的任意一个

```typescript
const canAccess = useHasAnyPermission(['hr:employee:view', 'hr:employee:edit']);
```

### 4. useHasAllPermissions(permissions: string[])
检查用户是否拥有指定权限中的所有

```typescript
const canManage = useHasAllPermissions(['hr:employee:view', 'hr:employee:edit', 'hr:employee:delete']);
```

### 5. useDataScope()
获取用户的数据权限范围

```typescript
const dataScope = useDataScope();
// 返回: 'ALL' | 'DEPT' | 'DEPT_ONLY' | 'SELF'
```

## 完整页面示例：员工管理

```typescript
import React, { useState } from 'react';
import { Table, Button, Space, Popconfirm, message, Card } from 'antd';
import { PlusOutlined } from '@ant-design/icons';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { useNavigate } from 'react-router-dom';
import { useHasPermission } from '@/hooks/usePermissions';
import NoPermission from '@/components/common/NoPermission';
import request from '@/utils/request';

const EmployeeListPage: React.FC = () => {
  const navigate = useNavigate();
  const queryClient = useQueryClient();

  // 权限检查
  const hasViewPermission = useHasPermission('hr:employee:view');
  const canCreate = useHasPermission('hr:employee:create');
  const canEdit = useHasPermission('hr:employee:edit');
  const canDelete = useHasPermission('hr:employee:delete');

  // 页面级权限控制
  if (!hasViewPermission) {
    return <NoPermission />;
  }

  // 获取员工列表
  const { data, isLoading } = useQuery({
    queryKey: ['employees'],
    queryFn: () => request.get('/hr/employees').then(res => res),
  });

  // 删除员工
  const deleteMutation = useMutation({
    mutationFn: (id: number) => request.delete(`/hr/employees/${id}`),
    onSuccess: () => {
      message.success('删除成功');
      queryClient.invalidateQueries({ queryKey: ['employees'] });
    },
  });

  const handleEdit = (record: any) => {
    navigate(`/hr/employees/${record.id}/edit`);
  };

  const handleDelete = (id: number) => {
    deleteMutation.mutate(id);
  };

  const columns = [
    { title: '工号', dataIndex: 'employeeNo', key: 'employeeNo' },
    { title: '姓名', dataIndex: 'name', key: 'name' },
    { title: '性别', dataIndex: 'gender', key: 'gender' },
    { title: '部门', dataIndex: ['org', 'name'], key: 'org' },
    {
      title: '操作',
      key: 'actions',
      render: (_, record) => (
        <Space>
          {canEdit && (
            <Button type="link" onClick={() => handleEdit(record)}>
              编辑
            </Button>
          )}
          {canDelete && (
            <Popconfirm
              title="确定删除?"
              onConfirm={() => handleDelete(record.id)}
              okText="确定"
              cancelText="取消"
            >
              <Button type="link" danger>删除</Button>
            </Popconfirm>
          )}
        </Space>
      ),
    },
  ];

  return (
    <Card
      title="员工管理"
      extra={
        canCreate && (
          <Button
            type="primary"
            icon={<PlusOutlined />}
            onClick={() => navigate('/hr/employees/create')}
          >
            新增员工
          </Button>
        )
      }
    >
      <Table
        columns={columns}
        dataSource={data?.items || []}
        rowKey="id"
        loading={isLoading}
        pagination={{
          total: data?.total || 0,
          pageSize: data?.pageSize || 10,
          current: data?.page || 1,
        }}
      />
    </Card>
  );
};

export default EmployeeListPage;
```

## 路由级权限控制（可选）

如果需要在路由层面控制权限，可以在 `src/router/index.tsx` 中添加：

```typescript
import { useHasPermission } from '@/hooks/usePermissions';

const ProtectedRoute = ({
  children,
  permission
}: {
  children: React.ReactNode;
  permission?: string;
}) => {
  const token = localStorage.getItem('token');
  const hasPermission = useHasPermission(permission || '');

  if (!token) {
    return <Navigate to="/login" replace />;
  }

  if (permission && !hasPermission) {
    return <Navigate to="/403" replace />;
  }

  return <>{children}</>;
};

// 使用示例
<Route path="/hr/employees" element={
  <ProtectedRoute permission="hr:employee:view">
    <EmployeeListPage />
  </ProtectedRoute>
} />
```
